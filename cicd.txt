1.2 Create GCP Service Account

gcloud iam service-accounts create github-actions

gcloud projects add-iam-policy-binding srinisre023 \
  --member="serviceAccount:github-actions@srinisre023.iam.gserviceaccount.com" \
  --role="roles/container.admin"

gcloud projects add-iam-policy-binding srinisre023 \
  --member="serviceAccount:github-actions@srinisre023.iam.gserviceaccount.com" \
  --role="roles/compute.admin"


Generate key:
gcloud iam service-accounts keys create key.json \
  --iam-account=github-actions@srinisre023.iam.gservice
